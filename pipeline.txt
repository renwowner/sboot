pipeline{
  agent any
    
    stages {
        stage('编译') {	    
            steps {
			    script{
					bat "cd test&&mvn clean compile"
				} 
            }
        }
		stage('打包') { 
            steps {
                script{
				bat "cd test&&mvn clean package"
				}
            }
        }
		stage('运行') {	    
            steps {
				script{

					bat "cd test/target&&java -jar test-0.0.1-SNAPSHOT.jar"
				}         
            }
        }
    }
}
